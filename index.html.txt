    function confirmKeyboard() {
      const value = parseFloat(keyboardValue.replace(',', '.'));
      
      if (isNaN(value) || value <= 0) {
        showNotification('Digite um valor v√°lido', 'error');
        return;
      }
      
      switch (keyboardTarget) {
        case 'others':
          addToCart('Outros', value);
          break;
          
        case 'received':
          receivedInput.value = value.toFixed(2).replace('.', ',');
          updateChange();
          break;
          
        case 'edit-quantity':
          if (editingIndex !== null) {
            cart[editingIndex].quantity = Math.max(1, parseInt(keyboardValue));
            renderCart();
            updateTotal();
          }
          break;
          
        case 'edit-price':
          if (editingIndex !== null) {
            cart[editingIndex].price = value;
            renderCart();
            updateTotal();
          }
          break;
      }
      
      closeKeyboard();
    }

    // Handle Keyboard Shortcuts
    function handleKeyboardShortcut(e) {
      if (e.key >= '0' && e.key <= '9') {
        e.preventDefault();
        inputDigit(e.key);
      } else if (e.key === ',' || e.key === '.') {
        e.preventDefault();
        inputComma();
      } else if (e.key === 'Backspace') {
        e.preventDefault();
        backspace();
      } else if (e.key === 'Enter') {
        e.preventDefault();
        confirmKeyboard();
      } else if (e.key === 'Escape') {
        e.preventDefault();
        cancelKeyboard();
      }
    }

    function setupPWA() {
      // Service Worker
      if ('serviceWorker' in navigator) {

          .then(registration => console.log('SW registered'))
          .catch(error => console.log('SW registration failed'));

    // Install App
    function installApp() {
      if (deferredPrompt) {
        deferredPrompt.prompt();
        deferredPrompt.userChoice.then((choiceResult) => {
          if (choiceResult.outcome === 'accepted') {
            console.log('User accepted the install prompt');
          }
          deferredPrompt = null;
          installPrompt.classList.remove('show');
        });
      }
    }

    // Auto-save feature
    setInterval(() => {
      if (cart.length > 0) {
        localStorage.setItem('currentCart', JSON.stringify({
          cart,
          total,
          paymentMethod,
          timestamp: Date.now()
        }));
      }
    }, 30000); // Save every 30 seconds

    // Load saved cart on startup
    function loadSavedCart() {
      const savedCart = localStorage.getItem('currentCart');
      if (savedCart) {
        const data = JSON.parse(savedCart);
        // Only load if less than 1 hour old
        if (Date.now() - data.timestamp < 3600000) {
          cart = data.cart;
          total = data.total;
          paymentMethod = data.paymentMethod;
          
          renderCart();
          updateTotal();
          
          if (paymentMethod) {
            selectPaymentMethod(paymentMethod);
          }
          
          showNotification('Carrinho anterior restaurado', 'success');
        }
      }
    }
  </script>
